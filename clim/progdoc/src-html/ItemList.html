001import java.util.HashMap;
002import java.util.Set;
003
004public class ItemList
005{
006    private HashMap<String, Item> aInventary;
007
008    /**
009     * Constructeur d'objets de classe ItemList
010     */
011    public ItemList()
012    {
013        this.aInventary = new HashMap<>();
014    }
015
016    public String getItemString(){
017        String vReturnString = "";
018        Set<String> vKeys = this.aInventary.keySet();
019        for(String vItem : vKeys)
020        {
021            vReturnString += " a "+vItem+"\n";
022        }
023        return vReturnString;
024    }
025    public void takeItem(final String pStringItem, final Item pItem){
026        this.aInventary.put(pStringItem, pItem);
027    }
028    public void dropItem(final String pStringItem){
029        this.aInventary.remove(pStringItem);
030    }
031    public Item getItem(final String pItem){
032        return this.aInventary.get(pItem);
033    }
034}
